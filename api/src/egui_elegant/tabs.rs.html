10		Self { tabs, selected }
11	}

14impl<'a> egui::Widget for ElegantTabs<'a> {
15	fn ui(self, ui: &mut egui::Ui) -> egui::Response {
16		let theme = ElegantTheme::get(ui.ctx());
17
18		let mut changed = false;
19		let mut response: Option<egui::Response> = None;
20
21		ui.horizontal(|ui| {
22			ui.spacing_mut().item_spacing.x = 24.0; // Spacing between tabs
23
24			for (i, &tab) in self.tabs.iter().enumerate() {
25				let is_selected = *self.selected == i;
26
27				let (rect, mut current_response) = ui.allocate_exact_size(
28					egui::vec2(
29						ui.painter()
30							.layout_no_wrap(
31								tab.to_string(),
32								egui::FontId::proportional(16.0),
33								theme.foreground,
34							)
35							.size()
36							.x,
37						32.0,
38					),
39					egui::Sense::click(),
40				);
41
42				if current_response.clicked() {
43					*self.selected = i;
44					changed = true;
45					current_response.mark_changed();
46				}
47
48				if current_response.hovered() {
49					ui.ctx().set_cursor_icon(egui::CursorIcon::PointingHand);
50				}
51
52				let text_color = if is_selected {
53					theme.foreground
54				} else if current_response.hovered() {
55					theme.foreground.linear_multiply(0.8)
56				} else {
57					theme.foreground.linear_multiply(0.6)
58				};
59
60				ui.painter().text(
61					rect.center(),
62					egui::Align2::CENTER_CENTER,
63					tab,
64					egui::FontId::proportional(16.0),
65					text_color,
66				);
67
68				if is_selected {
69					let line_y = rect.max.y - 2.0;
70					ui.painter().line_segment(
71						[
72							egui::pos2(rect.min.x, line_y),
73							egui::pos2(rect.max.x, line_y),
74						],
75						egui::Stroke::new(2.0, theme.primary),
76					);
77				}
78
79				if let Some(r) = response.as_mut() {
80					*r = r.union(current_response);
81				} else {
82					response = Some(current_response);
83				}
84			}
85		});
86
87		let mut res = response.unwrap_or_else(|| {
88			ui.allocate_response(egui::Vec2::ZERO, egui::Sense::hover())
89		});
90		if changed {
91			res.mark_changed();
92		}
93		res
94	}
95}
