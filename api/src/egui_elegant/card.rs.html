1use crate::theme::ElegantTheme;
2
3#[derive(Default)]
4pub struct Card {}
5
6impl Card {
7	pub fn new() -> Self {
8		Self::default()
9	}
10
11	pub fn show<R>(
12		self,
13		ui: &mut egui::Ui,
14		add_contents: impl FnOnce(&mut egui::Ui) -> R,
15	) -> egui::InnerResponse<R> {
16		let theme = ElegantTheme::get(ui.ctx());
17
18		egui::Frame::new()
19            .fill(theme.secondary.linear_multiply(0.3)) // faint background
20            .stroke(egui::Stroke::new(theme.spacing.border_width, theme.border))
21            .inner_margin(theme.spacing.card_inner_margin)
22            .corner_radius(egui::CornerRadius::same(theme.spacing.corner_radius as u8))
23            .show(ui, add_contents)
24	}
25}
