1use crate::{badge::ElegantBadge, theme::Variant, traits::Elegant};
2
3pub struct ElegantTagInput<'a> {
4	tags: &'a mut Vec<String>,
5	current_text: &'a mut String,
6}
7
8impl<'a> ElegantTagInput<'a> {
9	pub fn new(tags: &'a mut Vec<String>, current_text: &'a mut String) -> Self {
10		Self { tags, current_text }
11	}
12}
13
14impl<'a> egui::Widget for ElegantTagInput<'a> {
15	fn ui(self, ui: &mut egui::Ui) -> egui::Response {
16		let mut response: Option<egui::Response> = None;
17
18		ui.horizontal_wrapped(|ui| {
19			let mut to_remove = None;
20			for (i, tag) in self.tags.iter().enumerate() {
21				if ui
22					.add(ElegantBadge::new(tag).variant(Variant::Secondary))
23					.clicked()
24				{
25					to_remove = Some(i);
26				}
27			}
28
29			if let Some(i) = to_remove {
30				self.tags.remove(i);
31			}
32
33			let input_resp = ui.add(
34				egui::TextEdit::singleline(self.current_text)
35					.hint_text("Add a tag...")
36					.margin(egui::vec2(8.0, 8.0)),
37			);
38
39			if input_resp.lost_focus() && ui.input(|i| i.key_pressed(egui::Key::Enter)) {
40				let text = self.current_text.trim();
41				if !text.is_empty() {
42					self.tags.push(text.to_string());
43					self.current_text.clear();
44					input_resp.request_focus();
45				}
46			}
47
48			response = Some(input_resp);
49		});
50
51		// Render the bounding box manually if needed, but egui's wrapping horizontal
52		// layout works decently. We'll just return the TextEdit's response.
53		response.unwrap()
54	}
55}
